def create_clip_context_pair(row):
    clip_name = row['clip_name']
    if 'bar' in row['context_word']:
        return 'BAR-' + clip_name
    elif 'videogame' in row['context_word']:
        return 'VIDEOGAME-' + clip_name
    elif 'concert' in row['context_word']:
        return 'CONCERT-' + clip_name
    elif 'movie' in row['context_word']:
        return 'MOVIE-' + clip_name
    else:
        return 'NO_MATCH'

dataMIMC['clip_context_PAIR'] = dataMIMC.apply(create_clip_context_pair, axis=1)

Text Preprocessing

Different NLP models require different levels of text filtering to perform optimally.

BERT — Benefits from minimal preprocessing because:

  • Understands context and handles common words well
  • Has its own tokenisation and handles word variations
  • Only needs custom domain-specific stop words removed

TF-IDF — Needs more aggressive preprocessing because:

  • Dosen’t understand context as well
  • Common stop words add noise
  • Needs lemmatisation to group related terms

We will produce two levels of text preprocessing to save out for later analyses. All text will be manually spell checked.

from spellchecker import SpellChecker
spell = SpellChecker()
def find_potential_misspellings_with_context(row):
    misspellings_with_context = []
    text = row['descr_THOUGHT.text']
    if isinstance(text, str):
        # Tokenise the raw text
        tokens = word_tokenize(text)
        for word in tokens:
            # Clean the word for spell checking (lowercase and remove punctuation)
            cleaned_word = re.sub(r'[^a-zA-Z0-9]', '', word.lower())
            if cleaned_word and cleaned_word not in spell:
                misspellings_with_context.append((cleaned_word, text))  # Store word and original text
    return misspellings_with_context

def apply_manual_corrections(text, mapping):
    if isinstance(text, str):
        # Clean the text (lowercase and remove punctuation) before applying corrections
        cleaned_text = re.sub(r'[^a-zA-Z0-9\s]', '', text.lower())
        words = cleaned_text.split()
        # Apply corrections
        corrected_words = [str(mapping.get(word, word)) for word in words]
        return " ".join(corrected_words)
    return str(text)

These are words we will remove from the text that won’t be accounted for in analyses.

We will define custom domain-specific stop words (e.g. music style, context cues, thought types), and get NLTK common English-language stop words.

Both preprocessing levels will have custom stop words removed, but only level 2 preprocessing for TF-IDF will has the NLTK stop words removed (as BERT understands “the”, “is”, “and”, etc.).

Lemmatise

Lemmatising groups inflected word forms together by identifying the dictionary form (the lemma) to analyse them as a single item.

[e.g., improve/improving/improvements/improved/improver = improve]

Only level 2 preprocessing for TF-IDF will have lemmatisation (BERT is able to handle word variations).

Apply text preprocessing

LEVEL 1: Minimal preprocessing for BERT, only removing custom stop words from spell-checked MIMC descriptions.

LEVEL 2: Aggressive preprocessing for TF-IDF, removing custom stop words, NLTK stop words, lemmatising, converting to lowercase, and removing punctuation from the spell-checked MIMC descriptions.

def to_list(arg):
    return list(arg) if isinstance(arg, set) else arg

def preprocessing_lvl1a(text, stimuli_stop_words, thought_stop_words):
    """
    Level 1a: Minimal preprocessing for BERT
    - Only removes custom domain-specific stop words
    - Preserves original word forms (no lemmatisation)
    - Keeps punctuation and capitalization for BERT's tokeniser
    """
    if isinstance(text, str):
        custom_stop_words = set(to_list(stimuli_stop_words) + to_list(thought_stop_words))
        tokens = word_tokenize(text)
        filtered_tokens = []
        for word in tokens:
            if word.lower() not in custom_stop_words:
                filtered_tokens.append(word)
        return " ".join(filtered_tokens)
    return text

def preprocessing_lvl1b(text, stimuli_stop_words):
    """
    Same as Level 1a but doesn't remove thought-related stop words
    """
    if isinstance(text, str):
        custom_stop_words = set(to_list(stimuli_stop_words))
        tokens = word_tokenize(text)
        filtered_tokens = []
        for word in tokens:
            if word.lower() not in custom_stop_words:
                filtered_tokens.append(word)
        return " ".join(filtered_tokens)
    return text


def preprocessing_lvl2a(text, stimuli_stop_words, thought_stop_words, nltk_stop_words, lemmatiser, tag_map):
    """
    Level 2a: Aggressive preprocessing for TF-IDF
    - Removes all stop words (custom + NLTK)
    - Applies lemmatisation
    - Converts to lowercase
    - Removes punctuation
    """
    if isinstance(text, str):
        combined_stop_words = set(to_list(stimuli_stop_words) + to_list(thought_stop_words) + to_list(nltk_stop_words))
        lemmatized_stop_words = set()
        for sw in combined_stop_words:
            lemmatized_stop_words.add(lemmatiser.lemmatize(sw.lower(), wn.NOUN))
            lemmatized_stop_words.add(lemmatiser.lemmatize(sw.lower(), wn.VERB))

        tokens = word_tokenize(text)
        lemmatised_tokens = []
        for word, tag in pos_tag(tokens):
            cleaned_word = re.sub(r'[^a-zA-Z]', '', word.lower())
            if cleaned_word:
                lemmatised_word = lemmatiser.lemmatize(cleaned_word, tag_map[tag[0]])
                if lemmatised_word not in lemmatized_stop_words:
                    lemmatised_tokens.append(lemmatised_word)
        return " ".join(lemmatised_tokens)
    return text

def preprocessing_lvl2b(text, stimuli_stop_words, nltk_stop_words, lemmatiser, tag_map):
    """
    Same as Level 2a but doesn't remove thought-related stop words
    """
    if isinstance(text, str):
        combined_stop_words = set(to_list(stimuli_stop_words) + to_list(nltk_stop_words))
        lemmatized_stop_words = set()
        for sw in combined_stop_words:
            lemmatized_stop_words.add(lemmatiser.lemmatize(sw.lower(), wn.NOUN))
            lemmatized_stop_words.add(lemmatiser.lemmatize(sw.lower(), wn.VERB))

        tokens = word_tokenize(text)
        lemmatised_tokens = []
        for word, tag in pos_tag(tokens):
            cleaned_word = re.sub(r'[^a-zA-Z]', '', word.lower())
            if cleaned_word:
                lemmatised_word = lemmatiser.lemmatize(cleaned_word, tag_map[tag[0]])
                if lemmatised_word not in lemmatized_stop_words:
                    lemmatised_tokens.append(lemmatised_word)
        return " ".join(lemmatised_tokens)
    return text

Creating Combined combMIMC

Here, we are combining the preprocessed MIMC descriptions into singular aggregated documents grouped by stimuli condition i.e. specific clip and context pairing. This produces 64 document combinations (16 clips × 4 contexts).

Where the individual preprocessed MIMC descriptions can be analysed to measure semantic similarity between individual thought descriptions, combMIMC can be analysed to measure semantic similarity between aggregated representations of conditions.

def build_combMIMC(dataMIMC, text_col):
    """
    Aggregate individual MIMC descriptions by clip-context pair.
    Returns a DataFrame with one row per pair containing:
      - the aggregated NLP text column (joined with 'endofinput' separators)
      - text_original: the spell-checked original descriptions joined with ' | '
        separators, retained for human-readable reference in downstream analyses
      - metadata columns (clip_name, context_word, genre_code, etc.)
    """
    rows = []
    for idStimPair in np.unique(dataMIMC['clip_context_PAIR']):
        stimPairMask = dataMIMC['clip_context_PAIR'] == idStimPair
        filt = dataMIMC[stimPairMask]

        if len(filt) == 0:
            continue

        first_row  = filt.iloc[0]
        idClip     = first_row['clip_name']
        idContext  = first_row['context_word']

        # Aggregate NLP-preprocessed text (existing behaviour)
        joined_nlp = " endofinput ".join(str(v) for v in filt[text_col].values) + " endofinput "

        # Aggregate original spell-checked text for human-readable reference
        # Each participant's response is separated by ' | ' so the column remains
        # readable when inspecting the CSV or displaying pairs in analyses
        orig_col_src = 'descr_THOUGHT.text_spellchecked'
        joined_orig  = " | ".join(str(v) for v in filt[orig_col_src].values
                                  if isinstance(v, str) and v.strip())

        # Derive genre code from clip name prefix
        prefix = idClip[0:3]
        genre_map = {'80s': '80s', 'Jaz': 'Jaz', 'Met': 'Met', 'Ele': 'Ele'}
        genre_code = genre_map.get(prefix, 'UNK')

        rows.append({
            'context_word':      idContext,
            'genre_code':        genre_code,
            'clip_name':         idClip,
            'ClipContext_pair':  idStimPair,
            'GenreContext_pair': idStimPair[0:3] + "_" + prefix,
            text_col:            joined_nlp,
            'text_original':     joined_orig,
        })

    return pd.DataFrame(rows)
